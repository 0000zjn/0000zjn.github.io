Node.js 递归遍历文件夹并返回带路径的文件名数组

<!-- more -->
```javascript
const fs = require('fs');

const nameArr = traversingFoldersSync('./dist');
console.log(nameArr);

function traversingFoldersSync(path) {
  let fileArr = [];
  let pa = fs.readdirSync(path);
  pa.forEach(function (ele, index) {
    let info = fs.statSync(path + "/" + ele);
    if (info.isDirectory()) {
      // console.log("dir: " + ele)
      fileArr = fileArr.concat(traversingFoldersSync(path + "/" + ele));
    } else {
      // console.log("file: " +path + "/"+ ele);
      fileArr.push(path + "/"+ ele);
    }
  })
  return fileArr;
}
```
